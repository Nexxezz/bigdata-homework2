COMMANDS FOR DOCKER CONTAINERS:
docker network create hadoop-network
docker run -it --ZK_MYID=1 --hostname "namenode.cluster.com" --name "namenode" --network hadoop-network image_id
docker run -it --ZK_MYID=2 --hostname "standbynode.cluster.com" --name "standbynode" --network hadoop-network image_id
docker run -it --ZK_MYID=3 --hostname "datanode.cluster.com" --name "datanode" --network hadoop-network image_id

ALL NODES:
hadoop-daemon.sh start journalnode

ACTIVE NODE:
hdfs namenode -format
hadoop-daemon.sh start namenode

STANDBY NODE:
hdfs namenode -bootstrapStandby
hadoop-daemon.sh start namenode

ALL NODES:
zkServer.sh start

DATANODE:
hadoop-daemon.sh start datanode

ACTIVE && STANDBY NODES:
hdfs zkfc -formatZK
hadoop-daemon.sh start zkfc
hdfs haadmin -getServiceState namenode
hdfs haadmin -getServiceState standbynode
